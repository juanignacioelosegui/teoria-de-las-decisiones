            \[m + a(m-1+m) \geq 0\]
            \[m + a(2m-1) \geq 0\]
            \[m \geq -a(2m-1)\]
            \[m \geq -2ma+a\]
            \[m+2ma \geq a\]
            \[m(1+2a) \geq a\]
            \[m \geq \frac{a}{1+2a}\]
